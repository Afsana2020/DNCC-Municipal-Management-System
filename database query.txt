-- =====================================================
-- DATABASE: dbms
-- =====================================================

-- 1. users table (parent table)
CREATE TABLE `users` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `firstName` varchar(50) NOT NULL,
  `lastName` varchar(50) NOT NULL,
  `email` varchar(100) NOT NULL,
  `password` varchar(255) NOT NULL,
  `role` enum('admin','citizen') DEFAULT 'citizen',
  `created_at` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`)
) ENGINE=InnoDB AUTO_INCREMENT=24 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 2. assets table
CREATE TABLE `assets` (
  `Asset_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Asset_Type` varchar(50) NOT NULL,
  `Asset_Name` varchar(255) NOT NULL,
  `Asset_Condition` enum('Good','Fair','Poor','New Construction','Needs Maintenance','Under Maintenance') NOT NULL,
  `Expenses` decimal(15,2) unsigned NOT NULL,
  `Location` text NOT NULL,
  `Installation_Date` date NOT NULL,
  `Maintenance_Interval` int(10) unsigned NOT NULL,
  `Checking_Date` date DEFAULT NULL,
  `Last_Maintenance_Date` date DEFAULT NULL,
  `Created_At` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`Asset_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=38 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 3. projects table
CREATE TABLE `projects` (
  `Project_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Project_Name` varchar(255) NOT NULL,
  `Description` text DEFAULT NULL,
  `Budget` decimal(15,2) unsigned DEFAULT 0.00,
  `Project_Type` enum('Large','Routine','Urgent') NOT NULL,
  `Project_Director` varchar(100) DEFAULT NULL,
  `Start_Date` date DEFAULT NULL,
  `End_Date` date DEFAULT NULL,
  `Status` enum('Not Started','Active','Paused','Completed','Cancelled') NOT NULL DEFAULT 'Not Started',
  `Created_At` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`Project_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 4. packages table (depends on projects)
CREATE TABLE `packages` (
  `Package_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Project_ID` int(10) unsigned NOT NULL,
  `Package_Name` varchar(255) NOT NULL,
  `Description` text DEFAULT NULL,
  `Package_Type` enum('DPP','Maintenance') NOT NULL,
  `Team_Leader` varchar(100) DEFAULT NULL,
  `Budget` decimal(15,2) unsigned DEFAULT NULL,
  `Start_Date` date DEFAULT NULL,
  `End_Date` date DEFAULT NULL,
  `Status` enum('Not Started','Active','Paused','Completed','Cancelled') NOT NULL DEFAULT 'Not Started',
  `Created_At` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`Package_ID`),
  KEY `Project_ID` (`Project_ID`),
  CONSTRAINT `packages_ibfk_1` FOREIGN KEY (`Project_ID`) REFERENCES `projects` (`Project_ID`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 5. citizen_reports table (depends on users, assets, projects)
CREATE TABLE `citizen_reports` (
  `Report_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Report_type` varchar(100) DEFAULT NULL,
  `Report_Date` date DEFAULT NULL,
  `Description` text DEFAULT NULL,
  `Report_Image` longblob DEFAULT NULL,
  `user_id` int(10) unsigned DEFAULT NULL,
  `Status` enum('Submitted','Under Review','In Progress','Resolved','Closed') DEFAULT 'Submitted',
  `Admin_Response` text DEFAULT NULL,
  `Response_Date` date DEFAULT NULL,
  `Asset_ID` int(10) unsigned DEFAULT NULL,
  `Project_ID` int(10) unsigned DEFAULT NULL,
  PRIMARY KEY (`Report_ID`),
  KEY `user_id` (`user_id`),
  KEY `Asset_ID` (`Asset_ID`),
  KEY `Project_ID` (`Project_ID`),
  CONSTRAINT `citizen_reports_ibfk_1` FOREIGN KEY (`user_id`) REFERENCES `users` (`id`),
  CONSTRAINT `citizen_reports_ibfk_2` FOREIGN KEY (`Asset_ID`) REFERENCES `assets` (`Asset_ID`) ON DELETE SET NULL,
  CONSTRAINT `citizen_reports_ibfk_3` FOREIGN KEY (`Project_ID`) REFERENCES `projects` (`Project_ID`) ON DELETE SET NULL
) ENGINE=InnoDB AUTO_INCREMENT=14 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 6. citizens table (depends on citizen_reports, users)
CREATE TABLE `citizens` (
  `Citizen_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Citizen_name` varchar(100) DEFAULT NULL,
  `Address` text DEFAULT NULL,
  `Contact` varchar(20) DEFAULT NULL,
  `Report_ID` int(10) unsigned DEFAULT NULL,
  `user_id` int(10) unsigned DEFAULT NULL,
  PRIMARY KEY (`Citizen_ID`),
  KEY `Report_ID` (`Report_ID`),
  KEY `user_id` (`user_id`),
  CONSTRAINT `citizens_ibfk_1` FOREIGN KEY (`Report_ID`) REFERENCES `citizen_reports` (`Report_ID`) ON DELETE SET NULL,
  CONSTRAINT `citizens_ibfk_2` FOREIGN KEY (`user_id`) REFERENCES `users` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 7. zones table
CREATE TABLE `zones` (
  `Zone_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Zone_Name` mediumtext DEFAULT NULL,
  `Zone_Code` varchar(20) NOT NULL,
  `Created_At` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`Zone_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=11 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 8. project_zones table (junction table for projects and zones)
CREATE TABLE `project_zones` (
  `Project_Zone_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Project_ID` int(10) unsigned NOT NULL,
  `Zone_ID` int(10) unsigned NOT NULL,
  PRIMARY KEY (`Project_Zone_ID`),
  KEY `Project_ID` (`Project_ID`),
  KEY `Zone_ID` (`Zone_ID`),
  CONSTRAINT `project_zones_ibfk_1` FOREIGN KEY (`Project_ID`) REFERENCES `projects` (`Project_ID`) ON DELETE CASCADE,
  CONSTRAINT `project_zones_ibfk_2` FOREIGN KEY (`Zone_ID`) REFERENCES `zones` (`Zone_ID`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=10 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 9. maintenance table (depends on assets, citizen_reports, projects, packages, zones)
CREATE TABLE `maintenance` (
  `Maintenance_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `task_name` varchar(255) NOT NULL,
  `Task_type` enum('Construction','Repair','Maintenance','Restoration') NOT NULL,
  `Asset_ID` int(10) unsigned NOT NULL,
  `Cost` decimal(15,2) unsigned NOT NULL,
  `Start_Date` date NOT NULL,
  `End_Date` date DEFAULT NULL,
  `Description` text DEFAULT NULL,
  `Status` enum('Not Started','Active','Paused','Completed','Cancelled') NOT NULL DEFAULT 'Not Started',
  `Created_At` timestamp NOT NULL DEFAULT current_timestamp(),
  `Report_ID` int(10) unsigned DEFAULT NULL,
  `Project_ID` int(10) unsigned DEFAULT NULL,
  `Package_ID` int(10) unsigned DEFAULT NULL,
  `Zone_ID` int(10) unsigned DEFAULT NULL,
  PRIMARY KEY (`Maintenance_ID`),
  KEY `Asset_ID` (`Asset_ID`),
  KEY `Report_ID` (`Report_ID`),
  KEY `Project_ID` (`Project_ID`),
  KEY `Package_ID` (`Package_ID`),
  KEY `Zone_ID` (`Zone_ID`),
  CONSTRAINT `maintenance_ibfk_1` FOREIGN KEY (`Asset_ID`) REFERENCES `assets` (`Asset_ID`) ON DELETE CASCADE,
  CONSTRAINT `maintenance_ibfk_2` FOREIGN KEY (`Report_ID`) REFERENCES `citizen_reports` (`Report_ID`) ON DELETE SET NULL,
  CONSTRAINT `maintenance_ibfk_3` FOREIGN KEY (`Project_ID`) REFERENCES `projects` (`Project_ID`) ON DELETE SET NULL,
  CONSTRAINT `maintenance_ibfk_4` FOREIGN KEY (`Package_ID`) REFERENCES `packages` (`Package_ID`) ON DELETE SET NULL,
  CONSTRAINT `maintenance_ibfk_5` FOREIGN KEY (`Zone_ID`) REFERENCES `zones` (`Zone_ID`) ON DELETE SET NULL
) ENGINE=InnoDB AUTO_INCREMENT=85 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 10. budgets table (depends on assets, projects, packages)
CREATE TABLE `budgets` (
  `Budget_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Amount_Allocation` decimal(15,2) unsigned DEFAULT NULL,
  `Amount_Spent` decimal(15,2) unsigned DEFAULT NULL,
  `Asset_ID` int(10) unsigned DEFAULT NULL,
  `Project_ID` int(10) unsigned DEFAULT NULL,
  `Package_ID` int(10) unsigned DEFAULT NULL,
  PRIMARY KEY (`Budget_ID`),
  KEY `Asset_ID` (`Asset_ID`),
  KEY `Project_ID` (`Project_ID`),
  KEY `Package_ID` (`Package_ID`),
  CONSTRAINT `budgets_ibfk_1` FOREIGN KEY (`Asset_ID`) REFERENCES `assets` (`Asset_ID`) ON DELETE SET NULL,
  CONSTRAINT `budgets_ibfk_2` FOREIGN KEY (`Project_ID`) REFERENCES `projects` (`Project_ID`) ON DELETE SET NULL,
  CONSTRAINT `budgets_ibfk_3` FOREIGN KEY (`Package_ID`) REFERENCES `packages` (`Package_ID`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 11. workers table
CREATE TABLE `workers` (
  `Worker_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Worker_Name` varchar(100) NOT NULL,
  `Worker_Salary` decimal(15,2) unsigned NOT NULL,
  `Contact` varchar(20) DEFAULT NULL,
  `Designation` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`Worker_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=96 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 12. worker_assignments table (depends on workers, projects, packages, maintenance)
CREATE TABLE `worker_assignments` (
  `Assignment_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Worker_ID` int(10) unsigned NOT NULL,
  `Assignment_Type` enum('project','package','maintenance') NOT NULL,
  `Project_ID` int(10) unsigned DEFAULT NULL,
  `Package_ID` int(10) unsigned DEFAULT NULL,
  `Maintenance_ID` int(10) unsigned DEFAULT NULL,
  `Assigned_Date` datetime DEFAULT current_timestamp(),
  `Assigned_By` varchar(100) DEFAULT NULL,
  `Role` varchar(50) DEFAULT NULL,
  `Designation` varchar(100) DEFAULT NULL,
  `Status` enum('Active','Paused','Completed','Fired') NOT NULL DEFAULT 'Active',
  `Notes` text DEFAULT NULL,
  PRIMARY KEY (`Assignment_ID`),
  KEY `Worker_ID` (`Worker_ID`),
  KEY `Project_ID` (`Project_ID`),
  KEY `Package_ID` (`Package_ID`),
  KEY `Maintenance_ID` (`Maintenance_ID`),
  CONSTRAINT `fk_wa_maintenance` FOREIGN KEY (`Maintenance_ID`) REFERENCES `maintenance` (`Maintenance_ID`) ON DELETE SET NULL,
  CONSTRAINT `fk_wa_package` FOREIGN KEY (`Package_ID`) REFERENCES `packages` (`Package_ID`) ON DELETE SET NULL,
  CONSTRAINT `fk_wa_project` FOREIGN KEY (`Project_ID`) REFERENCES `projects` (`Project_ID`) ON DELETE SET NULL,
  CONSTRAINT `fk_wa_worker` FOREIGN KEY (`Worker_ID`) REFERENCES `workers` (`Worker_ID`) ON DELETE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=134 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 13. resources table (depends on maintenance)
CREATE TABLE `resources` (
  `Resource_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Resource_Type` varchar(100) DEFAULT NULL,
  `Quantity` int(10) unsigned NOT NULL,
  `Unit_Cost` decimal(10,2) unsigned DEFAULT NULL,
  `Maintenance_ID` int(10) unsigned DEFAULT NULL,
  PRIMARY KEY (`Resource_ID`),
  KEY `Maintenance_ID` (`Maintenance_ID`),
  CONSTRAINT `resources_ibfk_1` FOREIGN KEY (`Maintenance_ID`) REFERENCES `maintenance` (`Maintenance_ID`) ON DELETE SET NULL
) ENGINE=InnoDB AUTO_INCREMENT=97 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

-- 14. report_images table (depends on citizen_reports, maintenance, users)
CREATE TABLE `report_images` (
  `Image_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `Report_ID` int(10) unsigned DEFAULT NULL,
  `Maintenance_ID` int(10) unsigned DEFAULT NULL,
  `Image_Path` varchar(500) NOT NULL,
  `Uploaded_By` int(10) unsigned DEFAULT NULL,
  `Upload_Date` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`Image_ID`),
  KEY `Report_ID` (`Report_ID`),
  KEY `Maintenance_ID` (`Maintenance_ID`),
  KEY `Uploaded_By` (`Uploaded_By`),
  CONSTRAINT `report_images_ibfk_1` FOREIGN KEY (`Report_ID`) REFERENCES `citizen_reports` (`Report_ID`) ON DELETE CASCADE,
  CONSTRAINT `report_images_ibfk_2` FOREIGN KEY (`Maintenance_ID`) REFERENCES `maintenance` (`Maintenance_ID`) ON DELETE CASCADE,
  CONSTRAINT `report_images_ibfk_3` FOREIGN KEY (`Uploaded_By`) REFERENCES `users` (`id`) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
